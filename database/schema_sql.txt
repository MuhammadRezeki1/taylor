CREATE DATABASE IF NOT EXISTS jahit_buknining CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE jahit_buknining;

CREATE TABLE layanan (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nama VARCHAR(200) NOT NULL,
  deskripsi TEXT,
  harga_mulai INT DEFAULT 0,
  icon VARCHAR(100),
  urutan INT DEFAULT 0,
  aktif TINYINT(1) DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE testimoni (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nama VARCHAR(150) NOT NULL,
  kota VARCHAR(100),
  isi TEXT NOT NULL,
  bintang INT DEFAULT 5,
  foto VARCHAR(300),
  aktif TINYINT(1) DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE pesanan (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nama_pelanggan VARCHAR(150) NOT NULL,
  no_hp VARCHAR(20) NOT NULL,
  email VARCHAR(200),
  alamat TEXT,
  jenis_layanan VARCHAR(200),
  deskripsi_pesanan TEXT,
  status ENUM('baru','diproses','selesai','dibatalkan') DEFAULT 'baru',
  catatan_admin TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE kontak_masuk (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nama VARCHAR(150) NOT NULL,
  email VARCHAR(200),
  no_hp VARCHAR(20),
  pesan TEXT NOT NULL,
  sudah_dibaca TINYINT(1) DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE galeri (
  id INT AUTO_INCREMENT PRIMARY KEY,
  judul VARCHAR(200),
  deskripsi TEXT,
  url_gambar VARCHAR(500) NOT NULL,
  kategori VARCHAR(100) DEFAULT 'umum',
  urutan INT DEFAULT 0,
  aktif TINYINT(1) DEFAULT 1,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE blog (
  id INT AUTO_INCREMENT PRIMARY KEY,
  judul VARCHAR(400) NOT NULL,
  slug VARCHAR(400) NOT NULL UNIQUE,
  ringkasan TEXT,
  konten LONGTEXT,
  gambar_cover VARCHAR(500),
  penulis VARCHAR(150) DEFAULT 'Buk Nining',
  published TINYINT(1) DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE settings (
  id INT AUTO_INCREMENT PRIMARY KEY,
  `key` VARCHAR(100) NOT NULL UNIQUE,
  `value` TEXT,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
